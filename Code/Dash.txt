using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class Dash : MonoBehaviour
{
    public Dash dash;
    public float Boost = 50;
    public Rigidbody rb;
    public float sec = 5f;
    public Movement2D moveT;
    public Movement3D MoveT;

    void Start()
    {

        StartCoroutine(Coroutine());

    }

    IEnumerator Coroutine()
    {

        rb.AddForce(-Boost * Time.deltaTime, 0, 0, ForceMode.Impulse);

        yield return new WaitForSeconds(sec);

        dash.enabled = false;
    }

}
